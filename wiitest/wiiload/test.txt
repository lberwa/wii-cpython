
#include <gccore.h>
#include <fat.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
    
extern const unsigned char _binary_data_test2_start[];
extern const unsigned char _binary_data_test2_end[];

extern const unsigned char _binary_data_test1_start[];
extern const unsigned char _binary_data_test1_end[];

extern const unsigned char _binary_data_test3_start[];
extern const unsigned char _binary_data_test3_end[];


typedef struct {
    const unsigned char *start;
    const unsigned char *end;
    const char *path;
} FileEntry;

static void *xfb = NULL;
static GXRModeObj *rmode = NULL;


int main(int argc, char **argv)
{
    VIDEO_Init();
    WPAD_Init();

    rmode = VIDEO_GetPreferredMode(NULL);
    xfb = MEM_K0_TO_K1(SYS_AllocateFramebuffer(rmode));
    console_init(xfb, 20, 20, rmode->fbWidth, rmode->xfbHeight,
                 rmode->fbWidth * VI_DISPLAY_PIX_SZ);

    VIDEO_Configure(rmode);
    VIDEO_SetNextFramebuffer(xfb);
    VIDEO_SetBlack(FALSE);
    VIDEO_Flush();
    VIDEO_WaitVSync();

    printf("Start...
");

    if (!fatInitDefault())
    {
        printf("SD init fail
");
        return 1;
    }

   {
       const unsigned char *data = _binary_data_test2_start;
       size_t size = _binary_data_test2_end - _binary_data_test2_start;

       const char *path = "sd:/test2";


        FILE *f = fopen(path, "wb");
        if (!f)
            return 1;

        fwrite(data, 1, size, f);
        fclose(f);
    }



   {
       const unsigned char *data = _binary_data_test1_start;
       size_t size = _binary_data_test1_end - _binary_data_test1_start;

       const char *path = "sd:/test1";


        FILE *f = fopen(path, "wb");
        if (!f)
            return 1;

        fwrite(data, 1, size, f);
        fclose(f);
    }



   {
       const unsigned char *data = _binary_data_test3_start;
       size_t size = _binary_data_test3_end - _binary_data_test3_start;

       const char *path = "sd:/test/test3";


        FILE *f = fopen(path, "wb");
        if (!f)
            return 1;

        fwrite(data, 1, size, f);
        fclose(f);
    }




    return 0
}
